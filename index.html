        function showPage(page) {

            if (page === 'home') {

                home.classList.remove('hidden');

            } else {

                home.classList.add('hidden');

            }

            window.scrollTo(0, 0);

        }


        const waveCanvas = document.getElementById('waveCanvas');

        const wCtx = waveCanvas.getContext('2d');

        let waves = [

            { amplitude: 35, frequency: 0.005, speed: 0.015, color: 'rgba(255, 255, 255, 0.12)', offset: 0 },

            { amplitude: 50, frequency: 0.003, speed: 0.008, color: 'rgba(255, 255, 255, 0.06)', offset: 100 }

        ];


        function resize() {

            waveCanvas.width = window.innerWidth;

            waveCanvas.height = waveCanvas.parentElement.offsetHeight;

            if (renderer && camera) {

                const snareContainer = document.getElementById('snareContainer');

                camera.aspect = snareContainer.offsetWidth / snareContainer.offsetHeight;

                camera.updateProjectionMatrix();

                renderer.setSize(snareContainer.offsetWidth, snareContainer.offsetHeight);

            }

        }


        function animateWaves() {

            wCtx.clearRect(0, 0, waveCanvas.width, waveCanvas.height);

            waves.forEach(wave => {

                wCtx.beginPath();

                wCtx.strokeStyle = wave.color;

                wCtx.lineWidth = 1;

                const scale = window.innerWidth < 768 ? 0.6 : 1;

                for (let x = 0; x < waveCanvas.width; x++) {

                    const y = waveCanvas.height / 2 + Math.sin(x * wave.frequency + wave.offset) * (wave.amplitude * scale);

                    if (x === 0) wCtx.moveTo(x, y); else wCtx.lineTo(x, y);

                }

                wCtx.stroke();

                wave.offset += wave.speed;

            });

            requestAnimationFrame(animateWaves);

        }


        let scene, camera, renderer, snareGroup;

        function initThree() {

            const container = document.getElementById('snareContainer');

            if (!container) return;

            

            scene = new THREE.Scene();

            camera = new THREE.PerspectiveCamera(45, container.offsetWidth / container.offsetHeight, 0.1, 1000);

            camera.position.z = 10;


            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

            renderer.setSize(container.offsetWidth, container.offsetHeight);

            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

            container.appendChild(renderer.domElement);


            snareGroup = new THREE.Group();

            const wireframeMaterial = new THREE.MeshBasicMaterial({ color: 0x444444, wireframe: true, transparent: true, opacity: 0.4 });

            const stickMaterial = new THREE.MeshBasicMaterial({ color: 0x888888, wireframe: true, transparent: true, opacity: 0.6 });


            const shell = new THREE.Mesh(new THREE.CylinderGeometry(2, 2, 1.4, 16), wireframeMaterial);

            snareGroup.add(shell);


            const topHead = new THREE.Mesh(new THREE.CircleGeometry(2, 16), wireframeMaterial);

            topHead.rotation.x = -Math.PI / 2;

            topHead.position.y = 0.71;

            snareGroup.add(topHead);


            const stickGeo = new THREE.CylinderGeometry(0.04, 0.08, 5, 8);

            const stick1 = new THREE.Mesh(stickGeo, stickMaterial);

            stick1.rotation.z = Math.PI / 4;

            stick1.position.set(-1.5, 1.8, 0);

            const stick2 = new THREE.Mesh(stickGeo, stickMaterial);

            stick2.rotation.z = -Math.PI / 4;

            stick2.position.set(1.5, 1.8, 0);

            snareGroup.add(stick1, stick2);


            scene.add(snareGroup);

            animateThree();

        }


        function animateThree() {

            requestAnimationFrame(animateThree);

            if (snareGroup) {

                snareGroup.rotation.y += 0.005;

                snareGroup.rotation.x = Math.sin(Date.now() * 0.001) * 0.1;

            }

            renderer.render(scene, camera);

        }


        window.addEventListener('resize', resize);

        window.addEventListener('load', () => {

            resize();

            animateWaves();

            initThree();

        });
